### 풀이 1: 브루트 포스

```js
function solution(n) {
    let count = 1;
    
    // O(n^2)
    for (let i = 1; i < n; i++) {
        let sum = i;
        
        for (let j = i + 1; j < n; j++) {
            sum += j;
            
            if(sum > n) break;
            
            if(sum === n){
                count++;
                break;
            }
        }
    }
    
    return count;
}
```

### 풀이 2: 투 포인터

```js
function solution(n) {
    let count = 0;

    let start = 1
    let end = 1;
    let sum = 1;

    // O(n)
    while(start <= n){
        if(sum === n) count++;

        if (sum >= n) {
            sum -= start;  
            start++;
        } else {
            end++;
            sum += end; 
        }
    }

    return count;
}
```
